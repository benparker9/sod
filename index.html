#load plot level data, clean
df_plot = bs_data['plot_data']
df_plot.columns = df_plot.iloc[0]
df_plot = df_plot.drop(df_plot.index[0]).reset_index()
df_plot.columns = df_plot.columns.str.strip()
# isolate data for if burned and p.ram detection year
df_burnPlot = df_plot.filter(regex="Big Sur|burned|1st P.ram").iloc[:,0:-1]
# create column with binary 0-1 for if p.ram is present given value in p. ram detection year
df_burnPlot['p.ram'] = ((pd.to_numeric(df_burnPlot['1st P.ram Dection'], errors='coerce').fillna(0) != 0).astype(int))
#convert other columns into numeric, fill na's with 0
cols_to_convert = df_burnPlot.columns[1:]  # everything except first and last
df_burnPlot[cols_to_convert] = df_burnPlot[cols_to_convert].apply(pd.to_numeric, errors='coerce').fillna(0).astype(int)
# copy df for fire status, add fire status given values for burned 2008, 2016, 2020 binary
df_burn = df_burnPlot.copy()
col1 = df_burn.iloc[:, 1]
col2 = df_burn.iloc[:, 2]
col3 = df_burn.iloc[:, 3]
conditions = [
    (col1 == 1) & (col2 == 0) & (col3 == 0),  # dolan
    (col1 == 0) & (col2 == 1) & (col3 == 0),  # sober
    (col1 == 0) & (col2 == 0) & (col3 == 1),  # chalk
    (col1 == 1) & (col2 == 0) & (col3 == 1),  # dolan_chalk
    (col1 == 0) & (col2 == 1) & (col3 == 1),  # sober_chalk
    (col1 == 0) & (col2 == 0) & (col3 == 0)   # control
]
labels = ['dolan', 'sober', 'chalk', 'dolan_chalk', 'sober_chalk', 'control']
df_burn['group'] = np.select(conditions, labels, default='unknown')
df_burn.columns.values[0] = 'BS_plot'
# group for data visualization
plot_data = []
for name,df in df_burn.groupby('group'):
    counts = df['p.ram'].value_counts().reset_index() # need to count
    counts.columns = ['p.ram', 'count']
    counts['group'] = name
    if 0 not in counts['p.ram'].values:
        counts = pd.concat([counts, pd.DataFrame({'p.ram': [0], 'count': [0], 'group': [name]})])
    if 1 not in counts['p.ram'].values:
        counts = pd.concat([counts, pd.DataFrame({'p.ram': [1], 'count': [0], 'group': [name]})])
    plot_data.append(counts)
# Combine all into one dataframe
plot_df = pd.concat(plot_data, ignore_index=True)
plot_df = plot_df.sort_values(['group', 'p.ram']).reset_index(drop=True)
#multi plot to see groups
fig = plt.figure(constrained_layout=True, figsize=(10, 8))
gs = fig.add_gridspec(2, 2)
ax_top = fig.add_subplot(gs[0, :])
#plot for multiple disturbances
sns.barplot(data=plot_df, x='count', y='group',  hue='p.ram', ax=ax_top)
ax_top.set_title('a) Plot group counts and if prescence of p.ram')
#plot just comparing p.ram 
p = plot_df.groupby('p.ram').sum().reset_index()
ax_bl = fig.add_subplot(gs[1, 0])
sns.barplot(data=p,  x='count', hue='p.ram', ax=ax_bl)
ax_bl.set_title('b) P.ram Detections')
# plot for comparing fire/no-fire
ax_br = fig.add_subplot(gs[1, 1])
# Define fire/no-fire groups
fire_groups = ['dolan', 'sober', 'chalk', 'dolan_chalk', 'sober_chalk']
# Add a fire_status column
df_burn['fire_status'] = np.where(df_burn['group'].isin(fire_groups), 'Fire', 'No Fire')
# Compute total counts per fire_status
summary = df_burn.groupby('fire_status')['p.ram'].count().sort_index().reset_index()
summary.rename(columns={'p.ram': 'count'}, inplace=True)
order = ['No Fire', 'Fire']
sns.barplot(data=summary, y='fire_status', x='count', hue='fire_status', ax=ax_br, order=order)
ax_br.set_title('c) Fire vs. No-Fire')
